const MODEL = 'gemini-2.5-flash'; // 想換模型改這裡

const SYSTEM_PROMPT = `你是一位協助台灣體育師資培育（師培）的教學設計助手。老師或師培生給你一個他想教的體育單元，你要幫他長出「教學靈感」與「可寫進教案的教學目標」。

【學術立場一】來自 Pan, Ng & Legg (2026)《Beyond adaptation》(Sport, Education and Society)：
- 核心精神「從調整(adaptation)到重構(reframing)」：不要「先做給健全生、再調整給障礙生」，而是從設計起點就預期多元身體。
- Inclusion Spectrum 五類型經 CDT 重構：開放式→質問基礎假設；修正式→揭露而非補救（修改不是降低標準，是揭露標準本來就不中立）；平行式→打斷能力階層；分開式→暫時性標的介入（絕不變永久分軌）；身障運動→賦權空間（自成一格的運動傳統，不是次等替代）。
- 【絕對紅線】不可建議 disability simulation（讓健全生坐輪椅、蒙眼「體驗障礙的感覺」），研究證實會強化 ableism。

【學術立場二】UDL/DI/STEP 層級式整合架構（作者團隊進行中之未發表論述）：
這三者不是三套平行的清單，而是「由上而下」的三個運作層次：
- UDL＝課程設計層次（為什麼改、要達成什麼教育目的）
- DI＝教學實施層次（因應群體或個別差異的動態調整，例如分層任務、依準備度調整）
- STEP＝活動調整層次（用哪個旋鈕落地：Space空間／Task任務／Equipment器材／People人員）
關鍵：STEP 永遠「從屬於」某條 UDL 原則，是它的落地手段，不是與 UDL 並列的獨立原則。每個教學設計都要能說出「這是為哪條 UDL 原則服務、用哪個 STEP 條件落地」。
CDT 護欄（來自案例）：所有調整要刻意採「所有學生皆可選擇」取向，不是只針對障礙生單獨設計，否則會造成隱性排除。

【最重要的立場——UDL 是為全班而設計，不是為障礙生做個別調整】
使用者可能會勾選班上的障礙類別（採台灣特教分類）。但這些資訊的用途，是讓「為全班而設計」更貼近真實班級，絕不是要你「針對這幾類學生做特別調整」。你產出的每一個設計，都必須是「全班所有學生皆可選擇、皆能參與」的（universal），而不是「給某類障礙生的專屬版本」。如果你發現自己在寫「為了視障生，所以…」這種句子，就要停下來改寫成「這個設計讓所有學生（包含視障生）都能…」。這正是從「調整(adaptation)」到「重構(reframing)」的核心。

【最重要的觀念——UDL 三原則是「重疊的透鏡」，不是「互斥的抽屜」】
CAST 三原則（多元表徵、多元行動與表達、多元參與投入）從來不是要把每個設計歸類進三個抽屜其中一個。它們是三張互相重疊的透鏡：同一個設計（例如「有聲球」）往往同時觸動多條原則——從表徵看是用聲音承載力道訊息，從參與看是讓視障生能自主投入、獲得勝任感。差別只在「對某個學生、在某個當下」哪條路徑作用比較顯著。
因此，不要把靈感硬分進三格。要以「一個具體教學決策」為單位，標出它的「主要透鏡」（作用最顯著那一條），再列出它「也涉及」的其他原則，並帶出對應的 SDT 需求（自主／勝任／關係）。

【設計的多樣性要求——避免每次都長一樣】
1. designs 的 4-5 個設計，必須分散在不同的「課堂環節」（phase）：暖身、技能學習、遊戲比賽、評量——至少涵蓋其中三種，不可全部擠在同一環節。
2. STEP 四個旋鈕（Space／Task／Equipment／People）在整批設計中至少要用到三個，不可每個設計都只轉 Equipment。
3. 每個設計都要附「why」（設計用意）：一句話說清楚這個決策背後的教學意圖——它預期改變什麼、為誰打開哪條參與路徑。注意 why 不可寫成「為某類障礙生專屬」，要寫成全班取向。
4. 避免只給教科書式的老梗；優先給這個單元「特有」的、老師明天就能直接用的具體做法。
5. 若使用者提供「先前已生成過的方向」清單，必須刻意避開那些方向，給出明顯不同的設計。

【語氣】面向一般體育老師、師培生、體育系學生。口語、具體、可操作，不要學術腔。

【輸出格式】只回傳合法 JSON，不要任何前後文字、不要 markdown 反引號。結構：
{
  "unit": "單元名稱",
  "designs": [

      "why": "設計用意：這個決策預期改變什麼、打開哪條參與路徑（1句）",

// ── 前端呼叫的主函式 ──────────────────────
function generateIdeas(p) {
  p = p || {};
  const unit = (p.unit || '').trim();
  if (!unit) return { error: '請提供單元名稱' };

  const stageText = (p.stage && p.stage.trim())
    ? '教學對象學段：' + p.stage.trim() + '。請讓所有靈感、動作複雜度、認知層次、競賽性、器材選擇與教學目標的用詞，都符合這個學段的年齡適切性（age-appropriate）。'
    : '教學對象學段未指定，請用適用於多數學段的通用寫法，避免預設特定年齡。';

  const bits = [];
  if (p.learners && p.learners.length) bits.push(p.learners.join('、'));
  if (p.otherNeeds && p.otherNeeds.trim()) bits.push('其他說明：' + p.otherNeeds.trim());
  const learnerText = bits.length
    ? '這個班的多樣性樣貌包含（採台灣特教類別）：' + bits.join('；') + '。請注意：這些是用來讓「為全班而設計」更貼近真實班級，不是要你為這幾類學生做個別化的特別調整。所有設計都要是「全班皆可選」的。'
    : '這個班的學生在動作能力、感官功能、社會互動與過往運動經驗上都是多元的。請為全班的多樣性做設計。';

  const avoidText = (Array.isArray(p.avoid) && p.avoid.length)
    ? '以下方向先前已經生成過，請刻意避開、給出明顯不同的設計方向：' + p.avoid.slice(0, 30).join('、') + '。'
    : '';

  const userText = '我想教的單元是：「' + unit + '」。' + stageText + learnerText + avoidText +
    '請幫我長出教學靈感與教學目標。只輸出 JSON，不要 markdown 反引號。每個欄位文字精簡（detail 一句話、why 一句話、idea 一到兩句），確保 JSON 完整結束。';

  const r = callGemini_(userText);
  if (r.status === 0) return { error: '尚未設定 API key。請到 Apps Script「專案設定 → 指令碼屬性」新增 GEMINI_API_KEY。' };
  if (r.status !== 200) return { error: friendlyApiError_(r.status), detail: (r.body || '').slice(0, 400) };

  try {
    const data = JSON.parse(r.body);
    const cand = (data.candidates || [])[0] || {};
    const text = (cand.content && cand.content.parts)
      ? cand.content.parts.map(function (x) { return x.text || ''; }).join('')
      : '';
    const parsed = extractJson_(text);
    if (!parsed) {
      const truncated = cand.finishReason === 'MAX_TOKENS';
      return { error: truncated ? '內容太長被截斷，請再按一次生成' : '解析回傳失敗，請再按一次生成',
               detail: (text || r.body).slice(0, 300) };
    }
    return parsed;
  } catch (e) {
    return { error: '解析回傳失敗，請再按一次生成', detail: String(e).slice(0, 200) };
  }
}

// ── 呼叫 Gemini（429／5xx 自動重試一次） ──────────
function callGemini_(userText) {
  let r = callGeminiOnce_(userText);
  if (r.status === 429 || r.status >= 500) {
    Utilities.sleep(2500);
    r = callGeminiOnce_(userText);
  }
  return r;
}

function callGeminiOnce_(userText) {
  const key = PropertiesService.getScriptProperties().getProperty('GEMINI_API_KEY');
  if (!key) return { status: 0, body: '' };
  const payload = {
    system_instruction: { parts: [{ text: SYSTEM_PROMPT }] },
    contents: [{ role: 'user', parts: [{ text: userText }] }],
    generationConfig: {
      temperature: 0.8,
      maxOutputTokens: 8000,
      responseMimeType: 'application/json',
      thinkingConfig: { thinkingBudget: 0 }
    }
  };
  const res = UrlFetchApp.fetch(
    'https://generativelanguage.googleapis.com/v1beta/models/' + MODEL + ':generateContent',
    { method: 'post', contentType: 'application/json',
      headers: { 'x-goog-api-key': key },
      payload: JSON.stringify(payload), muteHttpExceptions: true }
  );
  return { status: res.getResponseCode(), body: res.getContentText() };
}

function friendlyApiError_(status) {
  if (status === 429) return 'Gemini 免費額度暫時用完（429）。等 30～60 秒再按一次生成即可。';
  if (status === 400 || status === 403) return 'API key 可能貼錯或被限制（' + status + '）。到指令碼屬性重設一次；若持續，到 AI Studio 重新產一組。';
  if (status >= 500) return 'Gemini 伺服器暫時出狀況（' + status + '），稍等再試。';
  return 'Gemini API 回傳 ' + status;
}

function extractJson_(raw) {
  if (!raw) return null;
  let t = raw.replace(/```json/gi, '').replace(/```/g, '').trim();
  try { return JSON.parse(t); } catch (e) {}
  const a = t.indexOf('{'), b = t.lastIndexOf('}');
  if (a !== -1 && b > a) { try { return JSON.parse(t.slice(a, b + 1)); } catch (e) {} }
  return null;
}
